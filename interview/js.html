function flatten(arr) {
  let temp = []
  for (let i = 0; i < arr.length; i++) {
    if (Array.isArray(arr[i])) {
      const t = flatten(arr[i])
      temp = [...temp, ...t]
    } else {
      temp.push(arr[i])
    }
  }
  return temp
}
